  Color _getAssistanceModeColor(AssistanceMode mode) {
    switch (mode) {
      case AssistanceMode.SELF_MANAGED: return Colors.green;
      case AssistanceMode.DEALER_ASSISTED: return Colors.blue;
      case AssistanceMode.FIELD_AGENT_ASSISTED: return Colors.purple;
      case AssistanceMode.HYBRID: return Colors.orange;
    }
  }

  Color _getSyncStatusColor(SyncStatus status) {
    switch (status) {
      case SyncStatus.SYNCED: return Colors.green.shade700;
      case SyncStatus.LOCAL_ONLY:
      case SyncStatus.QUEUED_FOR_SYNC: return Colors.blue.shade700;
      case SyncStatus.SYNCING: return Colors.amber.shade700;
      case SyncStatus.CONFLICTED:
      case SyncStatus.FAILED: return Colors.red.shade700;
    }
  }

  String _getSyncStatusLabel(SyncStatus status) {
    switch (status) {
      case SyncStatus.SYNCED: return 'Synced';
      case SyncStatus.LOCAL_ONLY: return 'Saved locally';
      case SyncStatus.QUEUED_FOR_SYNC: return 'Waiting';
      case SyncStatus.SYNCING: return 'Syncing';
      case SyncStatus.CONFLICTED: return 'Needs review';
      case SyncStatus.FAILED: return 'Failed';
    }
  }
}

🛠️ Part 3: Flutter Validation Implementation
1. OfflineFormValidator Service
dart
class OfflineFormValidator {
  final AppDatabase db;

  OfflineFormValidator(this.db);

  /// Validates farmer enrollment payload before sync queue insertion
  Future<String?> validateFarmerEnrollment({
    required String mobile,
    required AssistanceMode mode,
    required String villageId,
  }) async {
    // 1. Mobile format
    if (!RegExp(r'^\+?\d{10,15}$').hasMatch(mobile)) {
      return '📞 Enter valid phone number';
    }

    // 2. Duplicate check (local-only)
    final existing = await (db.select(db.farmers)
          ..where((t) => t.mobileNumber.equals(mobile) & t.tenantId.equals('CURRENT_TENANT')))
        .get();
    if (existing.isNotEmpty) return '👤 Farmer already enrolled';

    // 3. Enum validation (compile-time safe, but runtime guard)
    if (!AssistanceMode.values.contains(mode)) return '📋 Invalid mode selected';

    return null; // Valid
  }

  /// Validates parcel geometry & metadata locally
  Future<String?> validateParcelCreation({
    required double area,
    required List<List<double>> polygonCoords,
    required double gpsAccuracy,
  }) async {
    // 1. Area threshold
    if (area < 0.1) return '📏 Minimum area 0.1 ha required';

    // 2. Polygon validity (≥4 points, closed loop)
    if (polygonCoords.length < 4 || polygonCoords.first != polygonCoords.last) {
      return '🗺️ Draw complete boundary';
    }

    // 3. GPS accuracy
    if (gpsAccuracy > 15) return '📍 Improve GPS signal & try again';

    return null;
  }
}
