/// 🔄 SyncManager: Orchestrates offline queue processing, API batching, retry logic
class SyncManager {
  final AppDatabase db;
  final Dio dio;
  final String baseUrl;
  final String tenantId;
  final String deviceId;
  final Logger _logger = Logger();

  // Architectural constants (from Week 3 Governance)
  static const int _maxRetries = 10;
  static const Duration _initialBackoff = Duration(seconds: 30);
  static const Duration _maxBackoff = Duration(hours: 24);
  static const int _defaultBatchSize = 20;

  SyncManager({
    required this.db,
    required this.dio,
    required this.baseUrl,
    required this.tenantId,
    required this.deviceId,
  });

  /// 📥 Enqueue a mutation for background sync
  Future<void> enqueue({
    required EntityType entityType,
    required String localId,
    required OperationType operation,
    required Map<String, dynamic> payload,
    List<String>? dependencyIds,
    SyncPriority priority = SyncPriority.MEDIUM,
  }) async {
    final now = DateTime.now().toUtc();
    final eventId = const Uuid().v4();

    await db.into(db.syncQueue).insert(
          SyncQueueCompanion.insert(
            eventId: eventId,
            entityType: entityType,
            localId: localId,
            operation: operation,
            payloadJson: jsonEncode(payload),
            localTimestamp: now,
            version: 1,
            retryCount: 0,
            maxRetries: _maxRetries,
            nextRetryAfter: now, // Immediate attempt
            priority: priority,
            dependencyIds: dependencyIds?.join(','),
            createdAt: now,
          ),
          mode: InsertMode.insertOrIgnore, // Idempotency guarantee
        );

    _logger.i('📦 Enqueued sync event: $eventType -> $localId');
  }

  /// 🚀 Process pending sync queue (call from Workmanager or foreground)
  Future<SyncBatchResult> processQueue({int batchSize = _defaultBatchSize}) async {
    final now = DateTime.now().toUtc();
    
    // 1. Fetch eligible items (priority-ordered, ready for retry)
    final pending = await (db.select(db.syncQueue)
          ..where((t) => 
              t.syncStatus.equals(SyncStatus.QUEUED_FOR_SYNC) &
              t.nextRetryAfter.isSmallerOrEqualValue(now) &
              t.retryCount.isSmallerThanValue(_maxRetries))
          ..orderBy([
            (t) => OrderingTerm.asc(t.priority),
            (t) => OrderingTerm.asc(t.nextRetryAfter)
          ])
          ..limit(batchSize))
        .get();

    if (pending.isEmpty) return SyncBatchResult.empty();

    // 2. Filter out unresolved dependencies
    final eligible = await _filterByDependencies(pending);

    if (eligible.isEmpty) return SyncBatchResult.empty();

    // 3. Build API payload
    final batchPayload = eligible.map((item) => {
          'event_id': item.eventId,
          'entity_type': item.entityType.name,
          'local_id': item.localId,
          'operation': item.operation.name,
          'payload': jsonDecode(item.payloadJson),
          'local_timestamp': item.localTimestamp.toIso8601String(),
          'version': item.version,
          'dependency_ids': item.dependencyIds?.split(',').where((id) => id.isNotEmpty).toList(),
        }).toList();

    try {
      // 4. Dispatch to server
      final response = await dio.post(
        '$baseUrl/sync/events',
        options: Options(headers: {
          'X-Tenant-ID': tenantId,
          'X-Device-ID': deviceId,
          'Content-Type': 'application/json',
        }),
        data: {'events': batchPayload},
      );

      final serverResponse = response.data as Map<String, dynamic>;
      return await _handleServerResponse(serverResponse);
    } on DioException catch (e) {
      _logger.e('🌐 Sync network error: ${e.message}');
      await _markBatchRetry(eligible, e);
      return SyncBatchResult.failed(networkError: true);
    } catch (e) {
      _logger.e('❌ Sync unexpected error: $e');
      await _markBatchRetry(eligible, e);
      return SyncBatchResult.failed(networkError: false);
    }
  }

  /// 🧠 Dependency filtering: skip if referenced localIds are not synced
  Future<List<SyncQueueData>> _filterByDependencies(List<SyncQueueData> items) async {
    final eligible = <SyncQueueData>[];
    for (final item in items) {
      if (item.dependencyIds == null || item.dependencyIds!.isEmpty) {
        eligible.add(item);
        continue;
      }
      final depIds = item.dependencyIds!.split(',').where((id) => id.isNotEmpty);
      final unresolved = await (db.select(db.syncQueue)
            ..where((t) => t.localId.isIn(depIds) &
                ~t.syncStatus.equals(SyncStatus.SYNCED) &
                ~t.syncStatus.equals(SyncStatus.LOCAL_ONLY)))
          .get();

      if (unresolved.isEmpty) {
        eligible.add(item);
      } else {
        _logger.d('⏳ Dependency blocked: ${item.localId} waiting on ${unresolved.length} items');
        // Reschedule for later retry
        await (db.update(db.syncQueue)
              ..where((t) => t.eventId.equals(item.eventId)))
            .write(SyncQueueCompanion(
          nextRetryAfter: Value(_calculateNextRetry(item.retryCount)),
        ));
      }
    }
    return eligible;
  }

  /// 📨 Parse server response and update local DB state
  Future<SyncBatchResult> _handleServerResponse(Map<String, dynamic> response) async {
    final accepted = (response['accepted'] as List?)?.cast<String>() ?? [];
    final conflicts = (response['conflicts'] as List?)?.cast<Map<String, dynamic>>() ?? [];
    final failed = (response['failed'] as List?)?.cast<Map<String, dynamic>>() ?? [];

    // Mark accepted as SYNCED
    if (accepted.isNotEmpty) {
      await (db.update(db.syncQueue)
            ..where((t) => t.eventId.isIn(accepted)))
          .write(SyncQueueCompanion(
        syncStatus: Value(SyncStatus.SYNCED),
        lastError: Value(null),
      ));
    }

    // Route conflicts to ConflictResolver
    for (final conflict in conflicts) {
      await _resolveConflict(conflict);
    }

    // Retry failed items with backoff
    for (final failure in failed) {
      final eventId = failure['event_id'] as String;
      final error = failure['message'] ?? 'Unknown server error';
      final retryable = failure['retryable'] as bool? ?? true;

      final item = await (db.select(db.syncQueue)..where((t) => t.eventId.equals(eventId))).getSingle();
      
      if (retryable && item.retryCount < _maxRetries) {
        await (db.update(db.syncQueue)..where((t) => t.eventId.equals(eventId)))
            .write(SyncQueueCompanion(
          retryCount: Value(item.retryCount + 1),
          nextRetryAfter: Value(_calculateNextRetry(item.retryCount + 1)),
          lastError: Value(error),
        ));
      } else {
        await (db.update(db.syncQueue)..where((t) => t.eventId.equals(eventId)))
            .write(SyncQueueCompanion(
          syncStatus: Value(SyncStatus.FAILED),
          lastError: Value('MAX_RETRIES: $error'),
        ));
      }
    }

    return SyncBatchResult(
      acceptedCount: accepted.length,
      conflictCount: conflicts.length,
      failedCount: failed.length,
    );
  }

  /// 🔄 Exponential backoff + jitter
  DateTime _calculateNextRetry(int retryCount) {
    final delaySeconds = _initialBackoff.inSeconds * pow(2, retryCount).toInt();
    final capped = delaySeconds.clamp(0, _maxBackoff.inSeconds);
    final jitter = Random().nextInt(5); // 0-5 seconds jitter
    return DateTime.now().toUtc().add(Duration(seconds: capped + jitter));
  }

  /// ⚠️ Mark batch for retry on network failure
  Future<void> _markBatchRetry(List<SyncQueueData> items, dynamic error) async {
    final eventIds = items.map((i) => i.eventId);
    await (db.update(db.syncQueue)
          ..where((t) => t.eventId.isIn(eventIds)))
        .write(SyncQueueCompanion(
      retryCount: Value(items.first.retryCount + 1),
      nextRetryAfter: Value(_calculateNextRetry(items.first.retryCount + 1)),
      lastError: Value(error.toString()),
    ));
  }

  /// 🔀 Delegate conflict handling
  Future<void> _resolveConflict(Map<String, dynamic> conflictData) async {
    final resolver = ConflictResolver(db: db);
    await resolver.resolve(conflictData);
  }
}

class SyncBatchResult {
  final int acceptedCount;
  final int conflictCount;
  final int failedCount;
  final bool networkError;

  SyncBatchResult({
    this.acceptedCount = 0,
    this.conflictCount = 0,
    this.failedCount = 0,
    this.networkError = false,
  });

  factory SyncBatchResult.empty() => SyncBatchResult();
  factory SyncBatchResult.failed({required bool networkError}) =>
      SyncBatchResult(networkError: networkError);
}

/// 🛡️ ConflictResolver: Handles server-detected conflicts, updates local sync state, routes to manual review
class ConflictResolver {
  final AppDatabase db;
  final Logger _logger = Logger();

  ConflictResolver({required this.db});

  /// 🧩 Main resolution dispatcher
  Future<void> resolve(Map<String, dynamic> conflict) async {
    final eventId = conflict['event_id'] as String;
    final conflictType = conflict['conflict_type'] as String;
    final resolutionRequired = conflict['resolution_required'] as bool;
    final manualReviewUrl = conflict['manual_review_url'] as String?;

    _logger.w('⚠️ Conflict detected: $conflictType for event $eventId');

    // 1. Update local sync status
    await (db.update(db.syncQueue)
          ..where((t) => t.eventId.equals(eventId)))
        .write(SyncQueueCompanion(
      syncStatus: Value(SyncStatus.CONFLICTED),
      lastError: Value(jsonEncode(conflict)),
    ));

    // 2. Route based on conflict type
    switch (conflictType) {
      case 'version_mismatch':
        await _handleVersionMismatch(eventId, conflict);
        break;
      case 'geospatial_overlap':
        await _handleGeospatialConflict(eventId, conflict);
        break;
      case 'workflow_invalid':
        await _handleWorkflowConflict(eventId, conflict);
        break;
      default:
        await _routeToManualReview(eventId, manualReviewUrl ?? '');
    }

    // 3. Emit event for UI listeners (use Riverpod/Provider/BLoC)
    // emit(ConflictDetectedEvent(eventId, conflictType));
  }

  /// 🔄 Version mismatch: auto-merge or fetch server state
  Future<void> _handleVersionMismatch(String eventId, Map<String, dynamic> conflict) async {
    // Strategy: Fetch server version, apply semantic merge if safe, else flag
    _logger.i('🔄 Resolving version mismatch for $eventId');
    // Implement server fetch + merge logic here
    await _routeToManualReview(eventId, 'auto_merge_pending');
  }

  /// 🌍 Geospatial overlap: strict manual review
  Future<void> _handleGeospatialConflict(String eventId, Map<String, dynamic> conflict) async {
    _logger.e('🌍 Geospatial conflict requires manual review: $eventId');
    await _routeToManualReview(eventId, 'parcel_boundary_conflict');
  }

  /// 📐 Workflow invalid: reject local mutation, preserve for audit
  Future<void> _handleWorkflowConflict(String eventId, Map<String, dynamic> conflict) async {
    _logger.w('📐 Workflow transition rejected: $eventId');
    // Mark local record as failed but preserve payload for audit/replay
    await (db.update(db.syncQueue)
          ..where((t) => t.eventId.equals(eventId)))
        .write(SyncQueueCompanion(
      syncStatus: Value(SyncStatus.FAILED),
      lastError: Value('WORKFLOW_REJECTED: ${conflict['message']}'),
    ));
  }

dart
// main.dart or sync_initializer.dart
import 'package:workmanager/workmanager.dart';

void callbackDispatcher() {
  Workmanager().executeTask((task, inputData) async {
    if (task == 'syncTask') {
      final db = AppDatabase(/* your executor */);
      final dio = Dio(BaseOptions(baseUrl: 'https://api.yourdomain.com/v1'));
      final manager = SyncManager(
        db: db,
        dio: dio,
        tenantId: 'YOUR_TENANT_ID',
        deviceId: 'YOUR_DEVICE_UUID',
      );
      await manager.processQueue(batchSize: 20);
    }
    return true;
  });
}

void initBackgroundSync() {
  Workmanager().initialize(callbackDispatcher, isInDebugMode: true);
  Workmanager().registerPeriodicTask(
    'syncTask',
    'syncTask',
    frequency: const Duration(minutes: 15),
    constraints: Constraints(networkType: NetworkType.connected),
  );
}
