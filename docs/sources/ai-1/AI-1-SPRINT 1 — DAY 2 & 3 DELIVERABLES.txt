import 'package:drift/drift.dart';
import 'package:uuid/uuid.dart';

// Canonical Sync Status Enum
enum SyncStatus {
  LOCAL_ONLY,
  QUEUED_FOR_SYNC,
  SYNCING,
  CONFLICTED,
  SYNCED,
  FAILED
}

part 'platform_database.g.dart';

@DriftDatabase(tables: [
  Farmers, Parcels, CropCycles, StageInstances, CropActivities, SyncQueue
])
class AppDatabase extends _$AppDatabase {
  AppDatabase(QueryExecutor e) : super(e);

  @override
  int get schemaVersion => 1;

  @override
  MigrationStrategy get migration => MigrationStrategy(
        onCreate: (Migrator m) => m.createAll(),
        onUpgrade: (Migrator m, int from, int to) async {
          if (from < 2) {
            // Future migrations here
          }
        },
      );

  // 🔍 Quick sync helper
  Future<List<DataClass>> getPendingSyncItems() {
    return (select(syncQueue)
          ..where((t) => t.syncStatus.equals(SyncStatus.QUEUED_FOR_SYNC))
          ..orderBy([t => t.priority.asc()]))
        .get();
  }
}

class Farmers extends Table {
  TextColumn get id => text()();
  TextColumn get tenantId => text()();
  TextColumn get mobileNumber => text()();
  TextColumn get villageId => text()();
  TextColumn get assistanceMode => textEnum<AssistanceMode>();
  TextColumn get syncStatus => textEnum<SyncStatus>().withDefault(const Constant('LOCAL_ONLY'));
  IntColumn get syncVersion => integer().withDefault(const Constant(1));
  DateTimeColumn get createdAt => dateTime()();
  DateTimeColumn get updatedAt => dateTime()();

  // Audit fields
  TextColumn get auditActorId => text()();
  TextColumn get auditActorRole => textEnum<ActorRole>();
  DateTimeColumn get auditTimestamp => dateTime()();
  RealColumn get auditGpsLat => real().nullable()();
  RealColumn get auditGpsLng => real().nullable()();

        {mobileNumber, tenantId}, // Prevent duplicate enrollments per tenant

}

// 🔄 SYNC QUEUE (Persistent, survives reboots)
class SyncQueue extends Table {
  TextColumn get eventId => text()(); // Client-generated UUID for idempotency
  TextColumn get entityType => textEnum<EntityType>();
  TextColumn get localId => text()(); // Local DB ID
  TextColumn get operation => textEnum<OperationType>();

  DateTimeColumn get localTimestamp => dateTime()();
