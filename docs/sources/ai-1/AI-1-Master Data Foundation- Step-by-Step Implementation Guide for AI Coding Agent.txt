#!/usr/bin/env python3
"""
Fuzzy-match LGD villages to Census 2011 village directory.
Outputs: data/processed/villages_cross_referenced.csv
"""
import pandas as pd
from fuzzywuzzy import process, fuzz

LGD_VILLAGES = pd.read_csv("data/raw/lgd/villages.csv")
CENSUS_VILLAGES = pd.read_csv("data/raw/census/census_2011_villages.csv")

def match_village(lgd_row, census_df, threshold=85):
    # Fuzzy match by: village_name + district_name
    search_key = f"{lgd_row['canonical_name']}, {lgd_row['district_name']}"
    
    # Extract candidate census villages in same district
    candidates = census_df[census_df["district_name"] == lgd_row["district_name"]]["village_name"].tolist()
    
    if not candidates:
        return None
    
    # Fuzzy match
    match, score = process.extractOne(search_key, candidates, scorer=fuzz.token_sort_ratio)
    
    if score >= threshold:
        matched_census = census_df[
            (census_df["village_name"] == match) & 
            (census_df["district_name"] == lgd_row["district_name"])
        ].iloc[0]
        return {
            "lgd_id": lgd_row["id"],
            "census_village_code": matched_census["village_code"],
            "census_name": matched_census["village_name"],
            "match_score": score
        }
    return None

if __name__ == "__main__":
    results = []
    for _, lgd_row in LGD_VILLAGES.iterrows():
        match = match_village(lgd_row, CENSUS_VILLAGES)
        if match:
            results.append(match)
    
    matched_df = pd.DataFrame(results)
    merged = LGD_VILLAGES.merge(matched_df, left_on="id", right_on="lgd_id", how="left")
    merged.to_csv("data/processed/villages_cross_referenced.csv", index=False)
    print(f"✓ Cross-referenced {len(matched_df)} villages")

#!/usr/bin/env python3
"""
Join ArcGIS PIN dataset to villages by name + district fuzzy match.
Outputs: data/processed/villages_with_pins.csv
"""
import pandas as pd
import geopandas as gpd
from fuzzywuzzy import process, fuzz

VILLAGES = pd.read_csv("data/processed/villages_cross_referenced.csv")
PIN_DATA = gpd.read_file("data/raw/arcgis/pin_codes.geojson")  # From ArcGIS item

def match_pin_to_village(village_row, pin_gdf, threshold=80):
    # Search PIN points near village centroid
    village_point = Point(village_row["longitude"], village_row["latitude"])
    
    # Find PINs within 5km buffer
    nearby_pins = pin_gdf[pin_gdf.geometry.buffer(0.05).contains(village_point)]  # ~5km approx
    
    if nearby_pins.empty:
        # Fallback: fuzzy match by village name + district
        candidates = pin_gdf["village_name"].dropna().unique().tolist()
        match, score = process.extractOne(
            f"{village_row['canonical_name']}, {village_row['district_name']}",
            candidates,
            scorer=fuzz.token_sort_ratio
        )
        if score >= threshold:
            return pin_gdf[pin_gdf["village_name"] == match]["pin_code"].tolist()
        return []
    
    return nearby_pins["pin_code"].dropna().unique().tolist()

if __name__ == "__main__":
    VILLAGES["pin_codes"] = VILLAGES.apply(
        lambda row: match_pin_to_village(row, PIN_DATA), axis=1
    )
    VILLAGES.to_csv("data/processed/villages_with_pins.csv", index=False)
    print(f"✓ Enriched {VILLAGES['pin_codes'].apply(len).sum()} PIN codes")

#!/usr/bin/env python3
"""
Generate offline-ready compressed JSON cache for mobile app.
Outputs: mobile/assets/cache/geography_cache_v1.json.gz
"""
import json, gzip, pandas as pd
from pathlib import Path

VILLAGES = pd.read_csv("data/processed/villages_with_pins.csv")
DISTRICTS = pd.read_csv("data/processed/districts.csv")
STATES = pd.read_csv("data/processed/states.csv")

def build_cache():
    cache = {
        "version": "v1.0",
        "last_updated": pd.Timestamp.now().isoformat(),
        "states": STATES[["id", "lgd_code", "canonical_name", "aliases"]].to_dict("records"),
        "districts_by_state": {},
        "villages_by_district": {},
        "search_index": {
            "village_trigrams": {},  # FTS5 index data (optional: precompute)
            "village_phonetic": {}   # Soundex index (optional)
        }
    }
    
    # Group districts by state
    for state_id in STATES["id"].unique():
        state_districts = DISTRICTS[DISTRICTS["state_id"] == state_id]
        cache["districts_by_state"][state_id] = state_districts[
            ["id", "lgd_code", "canonical_name", "aliases", "tehsil_count", "village_count"]
        ].to_dict("records")
    
    # Group villages by district
    for district_id in DISTRICTS["id"].unique():
        district_villages = VILLAGES[VILLAGES["district_id"] == district_id]
        cache["villages_by_district"][district_id] = district_villages[
            ["id", "lgd_code", "canonical_name", "aliases", "pin_codes", "latitude", "longitude", "census_code"]
        ].to_dict("records")
    
    return cache

if __name__ == "__main__":
    cache = build_cache()
    
    # Compress with gzip
    output_path = Path("mobile/assets/cache/geography_cache_v1.json.gz")
    output_path.parent.mkdir(parents=True, exist_ok=True)
    
    with gzip.open(output_path, "wt", encoding="utf-8") as f:
        json.dump(cache, f, ensure_ascii=False, separators=(",", ":"))
    
    print(f"✓ Mobile cache generated: {output_path} ({output_path.stat().st_size / 1024:.1f} KB)")

#!/usr/bin/env python3
"""Block PRs if forbidden aliases appear in schema/code"""
import re, sys, pathlib

FORBIDDEN_PATTERNS = {
    "entity_names": re.compile(r"\b(field|plot|khet|cultivator)\b", re.IGNORECASE),
    "units": re.compile(r"\b(acres|quintal|local_currency)\b"),
}

def scan_file(filepath: pathlib.Path) -> list[str]:
    errors = []
    text = filepath.read_text()
    for category, pattern in FORBIDDEN_PATTERNS.items():
        if pattern.search(text):
            errors.append(f"🚫 Forbidden '{category}' in {filepath}")
    return errors

if __name__ == "__main__":
    paths = [pathlib.Path(p) for p in sys.argv[1:]]
    all_errors = []
    for p in paths:
        if p.suffix in (".py", ".dart", ".sql", ".json"):
            all_errors.extend(scan_file(p))
    
    if all_errors:
        print("\n❌ SEMANTIC VALIDATION FAILED:")
        for e in all_errors: print(f"  {e}")
        sys.exit(1)
    print("✅ Semantic validation passed")
