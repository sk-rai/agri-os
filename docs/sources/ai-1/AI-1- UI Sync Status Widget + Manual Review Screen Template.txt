import 'package:flutter/material.dart';

// Matches your Drift schema enum
enum SyncStatus {
  LOCAL_ONLY,
  QUEUED_FOR_SYNC,
  SYNCING,
  CONFLICTED,
  SYNCED,
  FAILED
}

/// 📊 Reusable sync status badge for list tiles, app bars, or cards
/// ✅ Rural UX Compliant: Icon + Color + ≤3 word label
class SyncStatusIndicator extends StatelessWidget {
  final SyncStatus status;
  final VoidCallback? onTap;
  final double iconSize;
  final double fontSize;

  const SyncStatusIndicator({
    super.key,
    required this.status,
    this.onTap,
    this.iconSize = 18,
    this.fontSize = 12,
  });

  @override
  Widget build(BuildContext context) {
    final config = _getStatusConfig(status);
    final child = GestureDetector(
      onTap: onTap,
      child: Row(
        mainAxisSize: MainAxisSize.min,
        children: [
          Icon(config.icon, size: iconSize, color: config.color),
          const SizedBox(width: 6),
          Text(config.label, style: TextStyle(color: config.color, fontWeight: FontWeight.w600, fontSize: fontSize)),
        ],
      ),
    );

    // Subtle pulse animation for active sync
    if (status == SyncStatus.SYNCING) {
      return TweenAnimationBuilder<double>(
        tween: Tween(begin: 1.0, end: 1.15),
        duration: const Duration(milliseconds: 800),
        curve: Curves.easeInOut,
        builder: (_, scale, child) => Transform.scale(scale: scale, child: child),
        child: child,
      );
    }
    return child;
  }

  _StatusConfig _getStatusConfig(SyncStatus status) {
    switch (status) {
      case SyncStatus.LOCAL_ONLY:
        return const _StatusConfig(Icons.check_circle_outline, Color(0xFF2E7D32), 'Saved locally');
      case SyncStatus.QUEUED_FOR_SYNC:
        return const _StatusConfig(Icons.cloud_upload_outlined, Color(0xFF1565C0), 'Waiting for internet');
      case SyncStatus.SYNCING:
        return const _StatusConfig(Icons.sync, Color(0xFF1565C0), 'Syncing...');
      case SyncStatus.CONFLICTED:
        return const _StatusConfig(Icons.warning_amber_rounded, Color(0xFFE65100), 'Needs attention');
      case SyncStatus.SYNCED:
        return const _StatusConfig(Icons.verified, Color(0xFF2E7D32), 'Synced');
      case SyncStatus.FAILED:
        return const _StatusConfig(Icons.error_outline, Color(0xFFC62828), 'Sync failed');
    }
  }
}

class _StatusConfig {
  final IconData icon;
  final Color color;
  final String label;
  const _StatusConfig(this.icon, this.color, this.label);
}

      ),
    ),
  );

  if (result == true) {
    // Refresh sync queue UI or re-trigger processQueue()
    syncManager.processQueue();
  }
}

// Add to sync_manager.dart
Future<void> markConflictResolved({
  required String eventId,
  required String resolutionAction,
  required DateTime resolvedAt,
}) async {
  await db.update(db.syncQueue)
    ..where((t) => t.eventId.equals(eventId))
    .write(SyncQueueCompanion(
      syncStatus: Value(SyncStatus.QUEUED_FOR_SYNC), // Re-queue with resolved payload
      lastError: Value('RESOLVED_BY_USER: $resolutionAction'),
    ));
  // Re-trigger sync for this item
  await processQueue(batchSize: 1);
}
