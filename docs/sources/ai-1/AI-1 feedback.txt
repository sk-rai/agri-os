3.2 Sync Protocol: State Machine
yaml
sync_protocol:
  states:
    local_only:
      description: Record exists only on device
      transitions:
        - queued_for_sync: when user action completes or app backgrounded

    queued_for_sync:
      description: Record in sync_queue, awaiting connectivity
      transitions:
        - syncing: when connectivity detected + sync job runs
        - failed: after max_retries exceeded

    syncing:
      description: Batch upload in progress
      transitions:
        - synced: server acknowledges with 200/201
        - conflicted: server returns 409 Conflict
        - failed: network error or 4xx/5xx (non-conflict)

    conflicted:
      description: Version/geospatial/workflow conflict detected
      transitions:
        - retry_pending: after manual resolution or auto-merge
        - failed: if resolution not possible

    synced:
      description: Server has authoritative version
      terminal: true

    failed:
      description: Unrecoverable error
      transitions:
        - retry_pending: if retryable error + retries remaining
      terminal: true if max_retries exceeded

  retry_policy:
    strategy: exponential_backoff
    initial_interval_seconds: 30
    max_interval_seconds: 86400 # 24 hours
    max_retries: 10
    jitter: true # Add randomness to avoid thundering herd

  conflict_detection:
    version_mismatch:
      trigger: server_version > client_version
      action: fetch_server_version + prompt_user_or_auto_merge

    geospatial_overlap:
      trigger: new_polygon overlaps existing by >5%
      action: route_to_manual_review_queue

    workflow_invalid:
      trigger: stage transition violates lifecycle_template
      action: reject_with_error + preserve_local_copy

  idempotency:
    mechanism: event_id deduplication
    server_behavior: ignore duplicate event_id with same payload
    client_behavior: regenerate event_id only if payload changes

        - Create parcel (depends on farmer) offline

    - name: retry_backoff

        - Simulate flaky network (50% failure rate)
        - Queue 10 sync events
        - Assert: exponential backoff observed; no duplicate server mutations; DLQ captures poison events

🔍 2. Semantic Registry Validator (scripts/validate_semantics.py)

#!/usr/bin/env python3
"""
Validates codebase against Canonical Semantic Registry v1.
Blocks PRs if forbidden terms appear in schema, API, or event definitions.
"""
import argparse, json, re, sys, pathlib

FORBIDDEN_PATTERNS = {
    "entity_names": re.compile(r"\b(field|plot|khet|cultivator|grower)\b", re.IGNORECASE),
    "enums": re.compile(r"\b(self_managed|dealer_managed|assisted)\b"),
    "units": re.compile(r"\b(acres|quintal|local_currency)\b"),
    "event_naming": re.compile(r"(send_|calculate_|run_|execute_)[a-z_]+\b")
}

CANONICAL_RULES = {
    "enum_values": {"SELF_MANAGED", "DEALER_ASSISTED", "FIELD_AGENT_ASSISTED", "HYBRID"},
    "unit_values": {"HECTARE", "KILOGRAM", "INR", "MILLIMETER"},
    "event_suffixes": [".v1", "_completed", "_created", "_updated", "_failed"]
}

def scan_file(filepath: pathlib.Path, registry: dict) -> list[str]:
    errors = []
    text = filepath.read_text()
    for category, pattern in FORBIDDEN_PATTERNS.items():
        if pattern.search(text):
            matches = pattern.findall(text)
            errors.append(f"🚫 Forbidden '{category}' found in {filepath.relative_to(pathlib.Path('.'))}: {matches}")
    return errors

def validate_enums(text: str, filepath: str) -> list[str]:
    # Simple check for canonical enum casing
    if "self_managed" in text.lower() and "SELF_MANAGED" not in text:
        return [f"⚠️ Enum 'self_managed' should be SCREAMING_SNAKE_CASE: SELF_MANAGED in {filepath}"]
    return []

def main():
    parser = argparse.ArgumentParser()
    parser.add_argument("--scan-paths", required=True, help="Comma-separated paths to scan")
    parser.add_argument("--registry", required=True, help="Path to semantic_registry_v1.json")
    args = parser.parse_args()

    paths = [pathlib.Path(p.strip()) for p in args.scan_paths.split(",")]
    registry = json.loads(pathlib.Path(args.registry).read_text())
    all_errors = []

    for path in paths:
        if path.is_file() and path.suffix in (".py", ".dart", ".json", ".sql", ".yml", ".yaml"):
            all_errors.extend(scan_file(path, registry))
            all_errors.extend(validate_enums(path.read_text(), str(path)))
        elif path.is_dir():
            for p in path.rglob("*"):
                if p.suffix in (".py", ".dart", ".json", ".sql", ".yml", ".yaml"):
                    all_errors.extend(scan_file(p, registry))
                    all_errors.extend(validate_enums(p.read_text(), str(p)))

    if all_errors:
        print("\n❌ SEMANTIC VALIDATION FAILED:")
        for err in all_errors:
            print(f"  {err}")
        sys.exit(1)
    else:
        print("\n✅ Semantic Registry v1 validation passed.")

if __name__ == "__main__":
    main()
